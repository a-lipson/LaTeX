\begin{problem}
Let $x,y\in \R^{n}$. \begin{enumerate}[label=\alph*)]
	\item Prove $2\left( {|x|}^2+{|y|}^2 \right) = {|x+y|}^2+{|x-y|}^2$.
	      \begin{proof}[Proof]
		      First, we will expand ${|x+y|}^2$,
		      \begin{align*}
			      {|x+y|}^2 & = \inp{x+y}{x+y}                              \\
			                & = \inp{x}{x+y}+\inp{y}{x+y}                   \\
			                & = \inp{x}{x}+\inp{x}{y}+\inp{y}{x}+\inp{y}{y} \\
			                & = {|x|}^2+2\inp{x }{y }+{|y|}^2.              \\
		      \end{align*}

		      We will do the same for ${|x-y|}^2$,
		      \begin{align*}
			      {|x-y|}^2 & = \inp{x-y}{x-y}                                      \\
			                & = \inp{x }{x-y }-\inp{y }{x-y }                       \\
			                & = \inp{x }{x }-\inp{x }{y }-\inp{y }{x }+\inp{y }{y } \\
			                & = {|x|}^2-2\inp{x }{y }+{|y|}^2.                      \\
		      \end{align*}

		      So, combining these, \[
			      {|x+y|}^2+{|x-y|}^2=2\left( {|x|}^2+{|y|}^2 \right)
			      ,\] as desired.
	      \end{proof}
	\item Prove the polarization identity, $\inp{x}{y}=\frac{{|x+y|}^2-{|x-y|}^2}{4}$.

\end{enumerate}
\end{problem}
